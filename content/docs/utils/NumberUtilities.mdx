# Number Utilities

Locale-aware number formatting powered by `Intl.NumberFormat`, `Intl.ListFormat`, and `Intl.PluralRules`. Zero dependencies.

```tsx
import { formatCurrency, formatCompactNumber, pluralize } from 'dandy-ui/utils';
```

---

## formatNumber

Formats a number using `Intl.NumberFormat`.

```tsx
formatNumber(1234.567);
// "1,234.567"

formatNumber(1234.567, { maximumFractionDigits: 2 });
// "1,234.57"
```

```ts
function formatNumber(
  value: number,
  options?: Intl.NumberFormatOptions,
  locale?: string
): string
```

---

## formatCurrency

Formats a number as currency.

```tsx
formatCurrency(1234.56, "USD");          // "$1,234.56"
formatCurrency(1234.56, "EUR", "de-DE"); // "1.234,56 €"
formatCurrency(1234.56, "JPY", "ja-JP"); // "¥1,235"
```

```ts
function formatCurrency(value: number, currency: string, locale?: string): string
```

---

## formatPercentage

Formats a number as a percentage. The value `0.25` becomes `"25%"`.

```tsx
formatPercentage(0.25);     // "25%"
formatPercentage(0.2567, 2); // "25.67%"
```

```ts
function formatPercentage(value: number, digits?: number, locale?: string): string
```

---

## formatCompactNumber

Formats large numbers in compact notation (1K, 1.2M, 1.5B).

```tsx
formatCompactNumber(1000);       // "1K"
formatCompactNumber(1234567);    // "1.2M"
formatCompactNumber(1500000000); // "1.5B"
```

```ts
function formatCompactNumber(value: number, locale?: string): string
```

---

## formatDigits

Formats a number with specified min/max fraction digits.

```tsx
formatDigits(1234.5, 2, 2);   // "1,234.50"
formatDigits(1234.567, 1, 3); // "1,234.567"
```

```ts
function formatDigits(value: number, min: number, max: number, locale?: string): string
```

---

## formatList

Formats an array into a conjunction list using `Intl.ListFormat`.

```tsx
formatList(["Alice", "Bob", "Charlie"]);
// "Alice, Bob, and Charlie"
```

```ts
function formatList(items: Array<string | number>, locale?: string): string
```

---

## clampNumber

Clamps a number between a minimum and maximum value.

```tsx
clampNumber(5, 0, 10);   // 5
clampNumber(15, 0, 10);  // 10
clampNumber(-5, 0, 10);  // 0
```

```ts
function clampNumber(value: number, min: number, max: number): number
```

---

## pluralize

Pluralizes a word based on count using `Intl.PluralRules`.

```tsx
pluralize("item", 1);              // "item"
pluralize("item", 3);              // "items"
pluralize("child", 2, "children"); // "children"
```

```ts
function pluralize(singular: string, count: number, plural?: string, locale?: string): string
```
